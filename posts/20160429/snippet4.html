function outRepeat(a){
        var hash=[],arr=[];
        for (var i = 0; i < a.length; i++) {
            hash[a[i]]!=null;
            if(!hash[a[i]]){
                arr.push(a[i]);
                hash[a[i]]=true;
            }
        }
        console.log(arr);
    }
    outRepeat([2,4,4,5,"a","a"]);//[2, 4, 5, "a"]
